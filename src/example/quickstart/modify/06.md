import top.katton.api.mod.*

// Farmer level 1: 1 emerald → 5 apples

    cost("minecraft:emerald", count = 1)
    result("minecraft:apple", count = 5)

    xp = 2
    priceMultiplier = 0.05f
}

    xp = 15
    priceMultiplier = 0.2f
